create or replace trigger book_changee
before update or delete on boook
for each row 
begin
    if deleting then
     insert into library_aud values(:old.bid,:old.author,:old.title,:old.Quantity,'DELETE');
    else 
      insert into library_aud values(:old.bid,:old.author,:old.title,:old.Quantity,'UPDATE');
    end if;
end;
